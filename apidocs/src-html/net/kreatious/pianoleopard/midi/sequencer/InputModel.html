001package net.kreatious.pianoleopard.midi.sequencer;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005import java.util.List;
006import java.util.Map;
007import java.util.Optional;
008import java.util.concurrent.CopyOnWriteArrayList;
009import java.util.function.Consumer;
010
011import javax.sound.midi.MidiDevice;
012import javax.sound.midi.MidiDevice.Info;
013import javax.sound.midi.MidiMessage;
014import javax.sound.midi.MidiUnavailableException;
015import javax.sound.midi.Receiver;
016
017import net.kreatious.pianoleopard.intervalset.IntervalSet;
018import net.kreatious.pianoleopard.midi.ParsedSequence;
019import net.kreatious.pianoleopard.midi.ParsedTrack;
020import net.kreatious.pianoleopard.midi.event.Event;
021import net.kreatious.pianoleopard.midi.event.EventFactory;
022import net.kreatious.pianoleopard.midi.event.EventPair;
023import net.kreatious.pianoleopard.midi.event.NoteEvent;
024import net.kreatious.pianoleopard.midi.event.PedalEvent;
025
026/**
027 * Model for the MIDI input keyboard, allows controllers to listen for events.
028 *
029 * @author Jay-R Studer
030 */
031public class InputModel implements AutoCloseable, ParsedTrack {
032    private Optional<MidiDevice> input = Optional.empty();
033    private final ReceiverImpl receiver = new ReceiverImpl();
034
035    private final IntervalSet<EventPair<NoteEvent>> notes = new IntervalSet<>();
036    private final IntervalSet<EventPair<PedalEvent>> pedals = new IntervalSet<>();
037
038    private final List<Consumer<? super Info>> inputDeviceListeners = new CopyOnWriteArrayList<>();
039    private final List<Consumer<? super Event>> inputListeners = new CopyOnWriteArrayList<>();
040
041    private InputModel(MidiDevice input) throws MidiUnavailableException {
042        setInputDevice(input);
043    }
044
045    /**
046     * Constructs a new {@link InputModel} with the specified initial state.
047     * <p>
048     * After construction, the input model is not connected to any actual
049     * devices. It is expected that the consumer will change the input device.
050     *
051     * @param outputModel
052     *            the output model to coordinate with
053     * @return a new instance of {@link InputModel}. The caller is responsible
054     *         for releasing the resource.
055     * @throws MidiUnavailableException
056     *             if the MIDI system is unavailable.
057     */
058    public static InputModel create(OutputModel outputModel) throws MidiUnavailableException {
059        final InputModel result = new InputModel(new InitialMidiDevice());
060        outputModel.addOpenListener(result::setCurrentSequence);
061        outputModel.addCurrentTimeListener(result.receiver::setCurrentTime);
062        return result;
063    }
064
065    private void setCurrentSequence(@SuppressWarnings("unused") ParsedSequence sequence) {
066        receiver.clear();
067    }
068
069    private final class ReceiverImpl implements Receiver, ParsedTrack {
070        private final Map<Object, NoteEvent> onNotes = new HashMap<>();
071        private final Map<Object, PedalEvent> onPedals = new HashMap<>();
072
073        private long currentTime;
074
075        private synchronized void setCurrentTime(long time) {
076            currentTime = time;
077        }
078
079        @Override
080        public synchronized void send(MidiMessage message, long timeStamp) {
081            EventFactory.create(message, currentTime).ifPresent(this::userPressedEvent);
082        }
083
084        private void userPressedEvent(Event event) {
085            if (event instanceof NoteEvent) {
086                userPressedEvent((NoteEvent) event, onNotes, notes);
087            } else if (event instanceof PedalEvent) {
088                userPressedEvent((PedalEvent) event, onPedals, pedals);
089            }
090            inputListeners.forEach(listener -> listener.accept(event));
091        }
092
093        private <K extends Event> void userPressedEvent(K event, Map<Object, K> onEvents,
094                IntervalSet<EventPair<K>> fullEvents) {
095            synchronized (fullEvents) {
096                final long eventTime = event.getTime();
097                if (event.isOn()) {
098                    onEvents.put(event.getSlot(), event);
099                } else {
100                    Optional.ofNullable(onEvents.remove(event.getSlot())).ifPresent(onEvent -> {
101                        fullEvents.put(onEvent.getTime(), eventTime, new EventPair<>(onEvent, event));
102                    });
103                }
104            }
105        }
106
107        @Override
108        public Iterable<EventPair<NoteEvent>> getNotePairs(long low, long high) {
109            return getPairs(low, high, onNotes, notes);
110        }
111
112        @Override
113        public Iterable<EventPair<PedalEvent>> getPedalPairs(long low, long high) {
114            return getPairs(low, high, onPedals, pedals);
115        }
116
117        private synchronized <K extends Event> Iterable<EventPair<K>> getPairs(long low, long high,
118                Map<Object, K> onEvents, IntervalSet<EventPair<K>> fullEvents) {
119            synchronized (fullEvents) {
120                final List<EventPair<K>> result = new ArrayList<>();
121                fullEvents.subSet(low, high).forEach(result::add);
122                onEvents.values().forEach(event -> result.add(new EventPair<>(event, event.createOff(currentTime))));
123                return result;
124            }
125        }
126
127        void clear() {
128            synchronized (notes) {
129                notes.clear();
130                onNotes.clear();
131            }
132            synchronized (pedals) {
133                pedals.clear();
134                onPedals.clear();
135            }
136        }
137
138        @Override
139        public void close() {
140            // Intentionally empty; this receiver holds no system resources
141        }
142    }
143
144    /**
145     * Reconnects the input to a different MIDI input device.
146     *
147     * @param input
148     *            the new input MIDI device to reconnect to
149     * @throws MidiUnavailableException
150     *             if the MIDI system is unavailable.
151     */
152    public void setInputDevice(MidiDevice input) throws MidiUnavailableException {
153        this.input.ifPresent(MidiDevice::close);
154        this.input = Optional.of(input);
155
156        input.open();
157        input.getTransmitter().setReceiver(receiver);
158        receiver.clear();
159        inputDeviceListeners.forEach(listener -> listener.accept(input.getDeviceInfo()));
160    }
161
162    /**
163     * Adds a listener to notify when the input device has changed.
164     *
165     * @param listener
166     *            the listener to add
167     */
168    public void addInputDeviceListener(Consumer<? super Info> listener) {
169        inputDeviceListeners.add(listener);
170    }
171
172    /**
173     * Adds a listener to notify when the user has pressed a key
174     *
175     * @param listener
176     *            the listener to add
177     */
178    public void addInputListener(Consumer<? super Event> listener) {
179        inputListeners.add(listener);
180    }
181
182    @Override
183    public void close() {
184        input.ifPresent(MidiDevice::close);
185    }
186
187    @Override
188    public Iterable<EventPair<NoteEvent>> getNotePairs(long low, long high) {
189        return receiver.getNotePairs(low, high);
190    }
191
192    @Override
193    public Iterable<EventPair<PedalEvent>> getPedalPairs(long low, long high) {
194        return receiver.getPedalPairs(low, high);
195    }
196}
