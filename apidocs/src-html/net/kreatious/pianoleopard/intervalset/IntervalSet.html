001package net.kreatious.pianoleopard.intervalset;
002
003import java.util.Iterator;
004import java.util.Optional;
005import java.util.function.Predicate;
006
007/**
008 * An ordered tree data structure for holding intervals. The user can
009 * efficiently find all intervals that overlap with a given interval or point.
010 * <p>
011 * This set does not support null keys or values, but does support duplicate
012 * intervals.
013 *
014 * @author Jay-R Studer
015 * @param <K>
016 *            the type of keys maintained by this set
017 * @param <V>
018 *            the type of mapped values
019 */
020public class IntervalSet<K extends Comparable<K>, V> implements Iterable<V> {
021    private Optional<Entry<K, V>> root = Optional.empty();
022    private int size;
023    private int modifications;
024
025    /**
026     * Removes all values from this set.
027     */
028    public void clear() {
029        root = Optional.empty();
030        size = 0;
031        modifications++;
032    }
033
034    /**
035     * Returns the size of this set.
036     *
037     * @return the number of values in this collection
038     */
039    public int size() {
040        return size;
041    }
042
043    /**
044     * Associates the specified interval with the specified value in this set.
045     * If this set previously contained a mapping for the interval, the old
046     * value is replaced. If the specified interval is invalid, nothing is
047     * inserted.
048     *
049     * @param low
050     *            the low end of the range the specified value to associate
051     * @param high
052     *            the high end of the range the specified value to associate
053     * @param value
054     *            value to be associated with the specified key
055     * @return the previous value associated with the interval, or empty if
056     *         there was no mapping for the interval.
057     * @throws NullPointerException
058     *             if the specified key or value is null
059     */
060    public Optional<V> put(K low, K high, V value) {
061        if (low.compareTo(high) > 0) {
062            return Optional.empty();
063        }
064
065        final Interval<K> key = new Interval<>(low, high);
066        if (!root.isPresent()) {
067            root = Optional.of(new Entry<>(key, value, Optional.empty()));
068            size = 1;
069            modifications++;
070            return Optional.empty();
071        }
072
073        final Optional<Entry<K, V>> parent = root.get().binarySearchInexact(key);
074        if (parent.filter(p -> p.getKey().compareTo(key) == 0).isPresent()) {
075            if (parent.get().addValue(value)) {
076                size++;
077                modifications++;
078                return Optional.empty();
079            }
080            return Optional.of(value);
081        }
082
083        final Entry<K, V> entry = new Entry<>(key, value, parent);
084        parent.get().insertNode(entry);
085        root = entry.rebalance(root);
086        size++;
087        modifications++;
088        return Optional.empty();
089    }
090
091    /**
092     * Removes the first value for the specified interval from this set that
093     * matches the given criteria.
094     *
095     * @param low
096     *            the low end of the range the specified value to associate
097     * @param high
098     *            the high end of the range the specified value to associate
099     * @param criteria
100     *            the value to delete
101     * @return the removed value, or empty if no value was removed
102     * @throws IllegalArgumentException
103     *             if {@code low} is greater than {@code high}
104     * @throws NullPointerException
105     *             if the specified key is null
106     */
107    public Optional<V> removeFirst(K low, K high, Predicate<? super V> criteria) {
108        final Optional<Entry<K, V>> entryToRemove = root.flatMap(entry -> entry.binarySearchExact(new Interval<>(low,
109                high)));
110        if (!entryToRemove.isPresent()) {
111            return Optional.empty();
112        }
113
114        final Entry<K, V> entry = entryToRemove.get();
115        final Optional<V> result = entry.getValues().stream().filter(criteria).findAny();
116        if (!result.isPresent()) {
117            return Optional.empty();
118        }
119
120        entry.getValues().remove(result.get());
121        if (entry.getValues().isEmpty()) {
122            root = entry.remove(root);
123        }
124        size--;
125        modifications++;
126        return result;
127    }
128
129    /**
130     * Returns an iterable read only view of the values in this set that
131     * overlaps the specified interval. If {@code low} and {@code high} are
132     * equal, the returned view contains the intervals containing a single
133     * point.
134     *
135     * @param low
136     *            low portion of the interval to retrieve
137     * @param high
138     *            high portion of the interval to retrieve
139     * @throws IllegalArgumentException
140     *             if {@code low} is greater than {@code high}
141     * @return a read only view of the portion of this set overlapping the
142     *         specified interval.
143     */
144    public Iterable<V> subSet(K low, K high) {
145        return new RangeIterable<>(this, low, high);
146    }
147
148    @Override
149    public Iterator<V> iterator() {
150        return new InOrderIterator<>(this);
151    }
152
153    int getModifications() {
154        return modifications;
155    }
156
157    Optional<Entry<K, V>> getRoot() {
158        return root;
159    }
160}
