20  /**
21   * Tests for {@link IntervalSet}'s range iterator
22   *
23   * @author Jay-R Studer
24   */
25  @RunWith(Parameterized.class)
26  public class RangeIteratorTest {
27      private static final IntervalSet<String> SET = new IntervalSet<>();
28  
29      /**
30       * The bounds for the test case
31       */
32      @Parameter(0)
33      public Interval interval;
34  
35      /**
36       * The values expected to be returned by the test
37       */
38      @Parameter(1)
39      public List<String> expectedValues;
40  
41      /**
42       * Provides a list of parameters to inject into the parameter fields
43       *
44       * @return a list of test cases
45       */
46      @Parameters(name = "interval [{0}] contains {1}")
47      public static List<Object[]> parameters() {
48          final List<Interval> intervals = new ArrayList<>();
49          final Random rnd = new Random(311);
50          for (int i = 0; i != 63; i++) {
51              final int low = rnd.nextInt(70);
52              final int high = low + rnd.nextInt(50);
53              intervals.add(new Interval(low, high));
54          }
55          intervals.forEach(interval -> SET.put(interval.getLow(), interval.getHigh(), interval.toString()));
56  
57          final List<Object[]> tests = new ArrayList<>();
58          final long max = intervals.stream().mapToLong(Interval::getHigh).max().getAsLong();
59          for (int i = 0; i <= max; i++) {
60              for (int intervalSize = 0; intervalSize != 10; intervalSize++) {
61                  final Interval testInterval = new Interval(i, i + intervalSize);
62  
63                  final List<String> expectedValues = intervals.stream()
64                          .filter(interval -> interval.containsInterval(testInterval)).map(Interval::toString)
65                          .collect(toList());
66                  tests.add(new Object[] { testInterval, expectedValues });
67              }
68          }
69          return tests;
70      }
71  
72      /**
73       * Tests that {@link IntervalSet#subSet} returns only the expected values
74       */
75      @Test
76      public void testSubSet() {
77          final Iterable<String> result = Lists.newArrayList(SET.subSet(interval.getLow(), interval.getHigh()));
78          expectedValues.forEach(value -> assertThat(result, hasItem(value)));
79          assertThat(result, iterableWithSize(expectedValues.size()));
80      }
81  }
