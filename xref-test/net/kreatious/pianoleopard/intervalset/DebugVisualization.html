51      static class OnTestFailureDebugClassRule<K extends Comparable<K>, V> implements TestRule {
52          final IntervalSet<K, V> set;
53          int failures;
54          private final DebugVisualization<K, V> visualization;
55  
56          OnTestFailureDebugClassRule(IntervalSet<K, V> set) {
57              this.set = set;
58              visualization = new DebugVisualization<>(set);
59          }
60  
61          @Override
62          public Statement apply(Statement base, Description description) {
63              return new Statement() {
64                  @Override
65                  public void evaluate() throws Throwable {
66                      base.evaluate();
67                      if (failures > 0) {
68                          visualization.setVisible(true);
69                      }
70                  }
71              };
72          }
73      }

75      static class OnTestFailureDebugRule<K extends Comparable<K>, V> implements TestRule {
76          private static final Color[] TEST_FAILURE_COLORS = new Color[] { Color.RED, Color.GREEN, Color.BLUE,
77                  Color.ORANGE, Color.PINK, Color.CYAN, Color.MAGENTA, Color.YELLOW };
78          private final OnTestFailureDebugClassRule<K, V> classRule;
79          private final Interval<K> testRange;
80  
81          OnTestFailureDebugRule(OnTestFailureDebugClassRule<K, V> classRule, Interval<K> testRange) {
82              this.classRule = classRule;
83              this.testRange = testRange;
84          }
85  
86          @Override
87          public Statement apply(Statement base, Description description) {
88              return new Statement() {
89                  @Override
90                  public void evaluate() throws Throwable {
91                      try {
92                          base.evaluate();
93                      } catch (final AssertionError e) {
94                          if (classRule.failures < TEST_FAILURE_COLORS.length && classRule.failures >= 0) {
95                              classRule.visualization.show(testRange.getLow(), testRange.getHigh(),
96                                      TEST_FAILURE_COLORS[classRule.failures++]);
97                          }
98                          throw e;
99                      }
100                 }
101             };
102         }
103     }

105     private class Decorations {
106         private final String searchInterval;
107         private final int number = ranges.size();
108         private String action = "";
109         private int step;
110         private Entry<K, V> previousPointer = set.getRoot().get();
111         private Entry<K, V> pointer = set.getRoot().get();
112         private final Color color;
113         private final Set<Entry<K, V>> expected;
114         private final Set<Entry<K, V>> visited = new HashSet<>();
115 
116         Decorations(K low, K high, Color color) {
117             searchInterval = "[" + low + ", " + high + "]";
118             this.color = color;
119             final List<Entry<K, V>> children = new ArrayList<>();
120             visitChildren(set.getRoot().get(), children::add);
121             expected = children.stream().filter(child -> child.getKey().containsInterval(new Interval<>(low, high)))
122                     .collect(toSet());
123         }
124     }

126     private class TreeDrawingPanel extends JPanel {
127         private static final long serialVersionUID = -8045457048224352208L;
128 
129         private TreeDrawingPanel() {
130             final Map<Entry<K, V>, Rectangle> positions = new HashMap<>();
131             calculateNodePositions(positions, Collections.singletonList(set.getRoot()), 1, maxLevel(set.getRoot()), 100);
132             setPreferredSize(new Dimension(positions.values().stream().map(rect -> rect.x + rect.width)
133                     .max(Integer::compare).get(), positions.values().stream().map(rect -> rect.y + rect.height)
134                     .max(Integer::compare).get()));
135 
136         }
137 
138         @Override
139         public void paint(Graphics g) {
140             super.paint(g);
141 
142             final Map<Entry<K, V>, Rectangle> positions = new HashMap<>();
143             calculateNodePositions(positions, Collections.singletonList(set.getRoot()), 1, maxLevel(set.getRoot()), 100);
144             positions.forEach((entry, position) -> {
145                 g.setColor(Color.BLACK);
146                 drawCenteredString(g, entry.toString(), position);
147 
148                 entry.getParent().ifPresent(
149                         parent -> {
150                             final Rectangle parentPosition = positions.get(parent);
151                             g.drawLine((int) parentPosition.getCenterX(), (int) parentPosition.getMaxY(),
152                                     (int) position.getCenterX(), (int) position.getMinY());
153                         });
154 
155                 g.drawRect(position.x, position.y, position.width, position.height);
156             });
157 
158             final Map<Entry<K, V>, Integer> numberTimesPainted = new HashMap<>();
159             ranges.values().forEach(
160                     decorator -> {
161                         final int timesPainted = numberTimesPainted.compute(decorator.pointer,
162                                 (key, value) -> value == null ? 0 : value + 1);
163 
164                         final Rectangle position = (Rectangle) positions.get(decorator.pointer).clone();
165                         g.setColor(decorator.color);
166                         position.x += position.width + 2;
167                         position.y += timesPainted * stringHeight(g, decorator.action);
168                         g.drawString(decorator.action, position.x, position.y);
169                         drawTraversal(g, decorator.previousPointer, decorator.pointer, decorator.number, positions);
170 
171                         decorator.expected.stream().map(positions::get).forEach(p -> {
172                             g.drawRect(p.x, p.y, p.width, p.height - decorator.number);
173                         });
174 
175                         decorator.visited.stream().map(positions::get).forEach(p -> {
176                             g.drawRect(p.x + 1, p.y + 1, p.width - 2, p.height - 2 - decorator.number);
177                         });
178                     });
179         }
180     }

182     DebugVisualization(IntervalSet<K, V> set) {
183         this.set = set;
184 
185         setModalityType(ModalityType.TOOLKIT_MODAL);
186         setResizable(true);
187         setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
188         setLayout(new FormLayout(new ColumnSpec[] { FormFactory.RELATED_GAP_COLSPEC, FormFactory.BUTTON_COLSPEC,
189                 FormFactory.RELATED_GAP_COLSPEC, ColumnSpec.decode("default:grow") }, new RowSpec[] {
190                 FormFactory.RELATED_GAP_ROWSPEC, FormFactory.DEFAULT_ROWSPEC, FormFactory.RELATED_GAP_ROWSPEC,
191                 RowSpec.decode("default:grow") }));
192 
193         final TreeDrawingPanel canvasTree = new TreeDrawingPanel();
194         final JButton step = new JButton("Step");
195         step.addActionListener(event -> {
196             ranges.forEach(new BiConsumer<Iterator<V>, Decorations>() {
197                 final Map<Set<V>, Entry<K, V>> entries = new IdentityHashMap<Set<V>, Entry<K, V>>() {
198                     private static final long serialVersionUID = 7676204394757535429L;
199                     {
200                         visitChildren(set.getRoot().get(), entry -> put(entry.getValues(), entry));
201                     }
202                 };
203 
204                 @Override
205                 public void accept(Iterator<V> iterator, DebugVisualization<K, V>.Decorations decorator) {
206                     if (iterator == null) {
207                         return;
208                     } else if (iterator.hasNext()) {
209                         decorator.previousPointer = decorator.pointer;
210                         decorator.pointer = entries.get(iterator.next());
211                         decorator.step++;
212                         decorator.action = decorator.searchInterval + " #" + decorator.step
213                                 + (decorator.expected.contains(decorator.pointer) ? "" : " (WRONG)");
214                         decorator.visited.add(decorator.pointer);
215                     } else {
216                         decorator.action = decorator.searchInterval + " #" + decorator.step + " (done)";
217                     }
218                 }
219             });
220             canvasTree.repaint();
221         });
222         add(step, "2, 2, 1, 1");
223         add(new JScrollPane(canvasTree), "1, 4, 4, 1");
224         pack();
225     }

234     private int maxLevel(Optional<Entry<K, V>> node) {
235         if (!node.isPresent()) {
236             return 0;
237         }
238 
239         return Math.max(maxLevel(node.get().getLeft()), maxLevel(node.get().getRight())) + 1;
240     }

242     private static int stringHeight(Graphics g, String s) {
243         return (int) g.getFontMetrics().getStringBounds(s, g).getHeight();
244     }

246     private static void drawCenteredString(Graphics g, String s, Rectangle position) {
247         final Rectangle2D bounds = g.getFontMetrics().getStringBounds(s, g);
248         g.drawString(s, (int) (position.getCenterX() - bounds.getCenterX()),
249                 (int) (position.getCenterY() - bounds.getCenterY()));
250     }

252     private void drawTraversal(Graphics g, Entry<K, V> start, Entry<K, V> end, int offsetY,
253             Map<Entry<K, V>, Rectangle> positions) {
254         final List<Entry<K, V>> successors = shortestPath(start, end);
255 
256         for (int i = 0; i < successors.size() - 1; i++) {
257             final Rectangle startPosition = positions.get(successors.get(i));
258             final Rectangle endPosition = positions.get(successors.get(i + 1));
259             g.drawLine(startPosition.x, (int) startPosition.getCenterY() + offsetY, endPosition.x,
260                     (int) endPosition.getCenterY() + offsetY);
261         }
262     }

264     private List<Entry<K, V>> shortestPath(Entry<K, V> start, Entry<K, V> end) {
265         final List<Entry<K, V>> result = new ArrayList<>();
266         for (int i = 0;; i++) {
267             if (shortestPath(start, end, i, result)) {
268                 Collections.reverse(result);
269                 return result;
270             }
271         }
272     }
273 
274     private boolean shortestPath(Entry<K, V> start, Entry<K, V> end, int depth, List<Entry<K, V>> result) {
275         if (depth < 0) {
276             return false;
277         }
278         if (start.equals(end)) {
279             result.add(end);
280             return true;
281         }
282         for (final Entry<K, V> next : Stream.of(start.getLeft(), start.getParent(), start.getRight())
283                 .filter(Optional::isPresent).map(Optional::get).collect(toList())) {
284             if (shortestPath(next, end, depth - 1, result)) {
285                 result.add(start);
286                 return true;
287             }
288         }
289         return false;
290     }

292     private void visitChildren(Entry<K, V> node, Consumer<Entry<K, V>> consumer) {
293         node.getLeft().ifPresent(next -> visitChildren(next, consumer));
294         consumer.accept(node);
295         node.getRight().ifPresent(next -> visitChildren(next, consumer));
296     }

298     private void calculateNodePositions(Map<Entry<K, V>, Rectangle> result, List<Optional<Entry<K, V>>> nodes,
299             int level, int maxLevel, int y) {
300         if (!nodes.stream().anyMatch(Optional::isPresent)) {
301             return;
302         }
303 
304         final int NODE_WIDTH = 76;
305         final int NODE_HEIGHT = 30;
306         final int SPACE_MULTIPLIER = NODE_WIDTH / 2;
307 
308         final List<Optional<Entry<K, V>>> newNodes = nodes
309                 .stream()
310                 .flatMap(
311                         node -> Stream.<Optional<Entry<K, V>>> of(node.flatMap(Entry::getLeft),
312                                 node.flatMap(Entry::getRight))).collect(toList());
313         final int betweenSpace = (int) Math.pow(2, maxLevel - level) * SPACE_MULTIPLIER;
314         int x = betweenSpace / 2;
315 
316         boolean overlapPossible = false;
317         for (int i = 0; i < nodes.size(); i++) {
318             if (nodes.get(i).isPresent()) {
319                 if (maxLevel == level && overlapPossible) {
320                     // prevent overlapping nodes at last level
321                     result.put(nodes.get(i).get(), new Rectangle(x, y + NODE_HEIGHT, NODE_WIDTH, NODE_HEIGHT));
322                     overlapPossible = false;
323                 } else {
324                     result.put(nodes.get(i).get(), new Rectangle(x, y, NODE_WIDTH, NODE_HEIGHT));
325                     overlapPossible = true;
326                 }
327             } else {
328                 overlapPossible = false;
329             }
330             x += betweenSpace;
331         }
332 
333         final int distanceToNextRow = betweenSpace / 16;
334         calculateNodePositions(result, newNodes, level + 1, maxLevel, y + distanceToNextRow + NODE_HEIGHT);
335     }
