39      public static void main(String[] args) throws Exception {
40          System.setProperty("java.util.concurrent.ForkJoinPool.common.parallelism", "256");
41  
42          final List<File> files = new ArrayList<>();
43          findMidis(new File("D:/Users/JStuder/Music/PianoCheetahSongs/Queue").toPath(), files);
44          final Map<Integer, Long> result = files.parallelStream().map(QueryAllMidis::accept)
45                  .collect(ConcurrentHashMap::new, (x, y) -> y.forEach((k, v) -> x.merge(k, v, Long::sum)), Map::putAll);
46          result.entrySet().stream().sorted(Comparator.comparing(x -> Long.MAX_VALUE - x.getValue()))
47                  .forEach(System.out::println);
48      }

50      private static final void findMidis(Path directory, List<File> files) throws IOException {
51          try (DirectoryStream<Path> stream = Files.newDirectoryStream(directory)) {
52              for (final Path path : stream) {
53                  final File file = path.toFile();
54                  if (file.isDirectory()) {
55                      findMidis(path, files);
56                  } else if (file.getName().toLowerCase().matches(".*(\\.mid|\\.midi)")) {
57                      files.add(file);
58                  }
59              }
60          }
61      }
62  
63      private static final Map<Integer, Long> accept(File file) {
64          try {
65              final ParsedSequence sequence = ParsedSequence.parseByTracks(MidiSystem.getSequence(file));
66              return sequence.getTracks().stream()
67                      .flatMap(track -> Lists.newArrayList(track.getNotePairs(0, Long.MAX_VALUE)).stream())
68                      .collect(groupingBy(EventPair::getChannel, summingLong((EventPair<?> x) -> x.getDuration())));
69          } catch (final InvalidMidiDataException | EOFException e) {
70              System.err.println(file);
71              return Collections.emptyMap();
72          } catch (final IOException e) {
73              System.err.println(file);
74              e.printStackTrace();
75              return Collections.emptyMap();
76          }
77      }
