9   import javax.sound.midi.MidiDevice;

14  import com.google.common.annotations.VisibleForTesting;
15  import com.jgoodies.forms.factories.FormFactory;

20  /**
21   * Provides a GUI element for selecting a MIDI Keyboard from a list of connected
22   * keyboards.
23   *
24   * @author Jay-R Studer
25   */
26  class KeyboardSelector {
27      /**
28       * Represents a row to be displayed within the combobox control
29       */
30      @VisibleForTesting
31      static class DeviceRow {
32          private final MidiDevice device;
33  
34          DeviceRow(MidiDevice device) {
35              this.device = device;
36          }
37  
38          MidiDevice getDevice() {
39              return device;
40          }
41  
42          @Override
43          public int hashCode() {
44              return device.hashCode();
45          }
46  
47          @Override
48          public boolean equals(Object obj) {
49              if (this == obj) {
50                  return true;
51              } else if (!(obj instanceof DeviceRow)) {
52                  return false;
53              }
54              return Objects.equals(((DeviceRow) obj).device, device);
55          }
56  
57          @Override
58          public String toString() {
59              return device.getDeviceInfo().getName();
60          }
61      }
62  
63      private final JComboBox<DeviceRow> keyboards = new JComboBox<>();
64      private final JPanel panel = new JPanel();
65      private final Predicate<? super MidiDevice> filter;
66      private final MidiDeviceFactory deviceFactory;
67  
68      /**
69       * Constructs a new {@link KeyboardSelector} with the specified label and
70       * filter.
71       * <p>
72       * The provided filter is used to determine if a keyboard should be
73       * included.
74       *
75       * @param label
76       *            the label text to display
77       * @param filter
78       *            the predicate determining if a keyboard should be included
79       * @param deviceFactory
80       *            the {@link MidiDeviceFactory} for obtaining available MIDI
81       *            devices
82       */
83      KeyboardSelector(String label, Predicate<? super MidiDevice> filter, MidiDeviceFactory deviceFactory) {
84          this.filter = filter;
85          this.deviceFactory = deviceFactory;
86  
87          panel.setLayout(new FormLayout(new ColumnSpec[] { FormFactory.BUTTON_COLSPEC,
88                  FormFactory.LABEL_COMPONENT_GAP_COLSPEC, ColumnSpec.decode("default:grow"), },
89                  new RowSpec[] { FormFactory.DEFAULT_ROWSPEC, }));
90          panel.add(new JLabel(label), "1, 1, right, default");
91          panel.add(keyboards, "3, 1, fill, default");
92  
93          reloadDevices();
94      }
95  
96      /**
97       * @return the {@link JPanel} associated with this control
98       */
99      JPanel getPanel() {
100         return panel;
101     }
102 
103     /**
104      * @return the current {@link MidiDevice} selected by this control.
105      */
106     Optional<MidiDevice> getSelectedDevice() {
107         return Optional.ofNullable((DeviceRow) keyboards.getSelectedItem()).map(DeviceRow::getDevice);
108     }
109 
110     /**
111      * @param device
112      *            the {@link MidiDevice} to set on this control
113      */
114     void setSelectedDevice(MidiDevice device) {
115         keyboards.setSelectedItem(new DeviceRow(device));
116     }
117 
118     /**
119      * Reloads the MIDI devices displayed by this control
120      */
121     void reloadDevices() {
122         final Optional<MidiDevice> selectedDevice = getSelectedDevice();
123         keyboards.removeAllItems();
124         Stream.of(deviceFactory.getMidiDevices()).filter(filter).map(DeviceRow::new).forEach(keyboards::addItem);
125         selectedDevice.ifPresent(this::setSelectedDevice);
126     }
127 
128     /**
129      * @return a new array containing the devices displayed by this control.
130      */
131     @VisibleForTesting
132     MidiDevice[] getDisplayedDevices() {
133         return IntStream.range(0, keyboards.getItemCount()).mapToObj(keyboards::getItemAt).map(DeviceRow::getDevice)
134                 .toArray(MidiDevice[]::new);
135     }
136 }
