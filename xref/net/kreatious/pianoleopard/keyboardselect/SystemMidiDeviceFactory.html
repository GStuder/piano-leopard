1   package net.kreatious.pianoleopard.keyboardselect;
2   
3   import java.util.Optional;
4   import java.util.stream.Stream;
5   
6   import javax.sound.midi.MidiDevice;
7   import javax.sound.midi.MidiDevice.Info;
8   import javax.sound.midi.MidiSystem;
9   import javax.sound.midi.MidiUnavailableException;
10  
11  /**
12   * Provides the MIDI devices connected to the current system.
13   *
14   * @author Jay-R Studer
15   */
16  public class SystemMidiDeviceFactory implements MidiDeviceFactory {
17      @Override
18      public MidiDevice[] getMidiDevices() {
19          return Stream.of(MidiSystem.getMidiDeviceInfo()).map(SystemMidiDeviceFactory::getMidiDevice)
20                  .filter(Optional::isPresent).map(Optional::get).toArray(MidiDevice[]::new);
21      }
22  
23      private static Optional<MidiDevice> getMidiDevice(Info info) {
24          try {
25              return Optional.of(MidiSystem.getMidiDevice(info));
26          } catch (final MidiUnavailableException e) {
27              return Optional.empty();
28          }
29      }
30  }
