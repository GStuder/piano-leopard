11  
12  import com.google.common.annotations.VisibleForTesting;
13  import com.jgoodies.forms.factories.FormFactory;
14  import com.jgoodies.forms.layout.ColumnSpec;
15  import com.jgoodies.forms.layout.FormLayout;

18  /**
19   * Provides the GUI for selecting the desired MIDI device
20   *
21   * @author Jay-R Studer
22   */
23  public class SelectKeyboardDialog {
24      private final KeyboardSelector input;
25      private final KeyboardSelector output;
26      private final JDialog dialog;
27      private Optional<Keyboard> keyboard;
28  
29      /**
30       * Constructs a new {@link SelectKeyboardDialog} for selecting the desired
31       * MIDI device.
32       *
33       * @param keyboard
34       *            the original {@link Keyboard} model to display, if applicable
35       * @param deviceFactory
36       *            the {@link MidiDeviceFactory} for obtaining available MIDI
37       *            devices
38       */
39      public SelectKeyboardDialog(Optional<Keyboard> keyboard, MidiDeviceFactory deviceFactory) {
40          this.keyboard = keyboard;
41          dialog = new JDialog();
42          dialog.setResizable(false);
43          dialog.setModalityType(ModalityType.DOCUMENT_MODAL);
44          dialog.setTitle("MIDI Keyboard Selection");
45          dialog.setLayout(new FormLayout(new ColumnSpec[] { FormFactory.RELATED_GAP_COLSPEC, FormFactory.BUTTON_COLSPEC,
46                  FormFactory.RELATED_GAP_COLSPEC, ColumnSpec.decode("default:grow"), FormFactory.RELATED_GAP_COLSPEC,
47                  FormFactory.BUTTON_COLSPEC, FormFactory.RELATED_GAP_COLSPEC, FormFactory.BUTTON_COLSPEC,
48                  FormFactory.RELATED_GAP_COLSPEC, }, new RowSpec[] { FormFactory.RELATED_GAP_ROWSPEC,
49                  FormFactory.DEFAULT_ROWSPEC, FormFactory.RELATED_GAP_ROWSPEC, FormFactory.DEFAULT_ROWSPEC,
50                  FormFactory.RELATED_GAP_ROWSPEC, FormFactory.DEFAULT_ROWSPEC, FormFactory.RELATED_GAP_ROWSPEC,
51                  FormFactory.DEFAULT_ROWSPEC, FormFactory.RELATED_GAP_ROWSPEC, }));
52  
53          dialog.add(new JLabel("Select which MIDI keyboard to use"), "2, 2, 7, 1");
54  
55          input = new KeyboardSelector("Input:", device -> device.getMaxTransmitters() != 0, deviceFactory);
56          keyboard.map(Keyboard::getInput).ifPresent(input::setSelectedDevice);
57          dialog.add(input.getPanel(), "2, 4, 7, 1");
58  
59          output = new KeyboardSelector("Output:", device -> device.getMaxReceivers() != 0, deviceFactory);
60          keyboard.map(Keyboard::getOutput).ifPresent(output::setSelectedDevice);
61          dialog.add(output.getPanel(), "2, 6, 7, 1");
62  
63          final JButton btnOk = new JButton("OK");
64          btnOk.addActionListener(event -> apply());
65          btnOk.addActionListener(event -> dialog.dispose());
66          dialog.add(btnOk, "6, 8");
67  
68          final JButton btnRefresh = new JButton("Refresh");
69          btnRefresh.setMnemonic(KeyEvent.VK_R);
70          btnRefresh.addActionListener(event -> input.reloadDevices());
71          btnRefresh.addActionListener(event -> output.reloadDevices());
72          dialog.add(btnRefresh, "2, 8");
73  
74          final JButton btnCancel = new JButton("Cancel");
75          btnCancel.addActionListener(event -> dialog.dispose());
76          dialog.add(btnCancel, "8, 8");
77  
78          dialog.pack();
79      }
80  
81      private void apply() {
82          keyboard = Optional.of(new Keyboard(input.getSelectedDevice().get(), output.getSelectedDevice().get()));
83      }
84  
85      /**
86       * @return the {@link JDialog} associated with this dialog
87       */
88      @VisibleForTesting
89      JDialog getDialog() {
90          return dialog;
91      }
92  
93      /**
94       * Shows the select keyboard dialog. This method blocks until a keyboard has
95       * been selected.
96       * <p>
97       * If a parent is not provided, the dialog will not be shown and this method
98       * returns immediately.
99       *
100      * @param parent
101      *            the parent component to layout this dialog relative to
102      * @return the selected keyboard, unless the dialog was cancelled
103      */
104     public Optional<Keyboard> showDialog(Optional<Component> parent) {
105         parent.ifPresent(dialog::setLocationRelativeTo);
106 
107         final Optional<Keyboard> oldKeyboard = keyboard;
108         input.reloadDevices();
109         output.reloadDevices();
110         dialog.setVisible(parent.isPresent());
111         return oldKeyboard.equals(keyboard) ? Optional.empty() : keyboard;
112     }
113 
114     /**
115      * @return the selected keyboard
116      */
117     @VisibleForTesting
118     Optional<Keyboard> getKeyboard() {
119         return keyboard;
120     }
121 }
