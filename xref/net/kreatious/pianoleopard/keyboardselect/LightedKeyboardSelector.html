3   import java.awt.Container;
4   import java.util.prefs.Preferences;
5   import java.util.stream.IntStream;
6   
7   import javax.swing.ImageIcon;
8   import javax.swing.JComboBox;
9   import javax.swing.JLabel;
10  import javax.swing.JPanel;
11  
12  import com.jgoodies.forms.factories.FormFactory;
13  import com.jgoodies.forms.layout.ColumnSpec;
14  import com.jgoodies.forms.layout.FormLayout;
15  import com.jgoodies.forms.layout.RowSpec;
16  
17  /**
18   * Provides a GUI element for selecting the options of a lighted keyboard
19   *
20   * @author Jay-R Studer
21   */
22  public class LightedKeyboardSelector {
23      /**
24       * The preference storage location for the navigation channel.
25       * <p>
26       * The channel is stored in the range 0..15.
27       */
28      public static final String NAV_CHANNEL_PREFERENCE = "navigationChannel";
29  
30      private final JComboBox<Integer> channels = new JComboBox<>(IntStream.rangeClosed(1, 16).boxed()
31              .toArray(Integer[]::new));
32  
33      /**
34       * Constructs a new {@link LightedKeyboardSelector} for selecting the
35       * options of a lighted keyboard
36       */
37      LightedKeyboardSelector() {
38      }
39  
40      /**
41       * Adds this selector to the specified container.
42       * <p>
43       * The container must have a JGoodies FormLayout with 1 row and 5 columns.
44       *
45       * @param container
46       *            the container to add to
47       * @param x
48       *            the column to add to
49       * @param y
50       *            the row to add to
51       * @param width
52       *            the number of columns to take up
53       */
54      void addToContainer(Container container, int x, int y, int width) {
55          final JPanel panel = new JPanel(new FormLayout(new ColumnSpec[] { ColumnSpec.decode("default:grow"),
56                  FormFactory.RELATED_GAP_COLSPEC, FormFactory.DEFAULT_COLSPEC, },
57                  new RowSpec[] { FormFactory.DEFAULT_ROWSPEC }));
58          final JLabel lblInfo = new JLabel(new ImageIcon(LightedKeyboardSelector.class.getResource("/help.png")));
59          lblInfo.setToolTipText("<html>Navigation channels are used by MIDI keyboards with lighted keys.<br>"
60                  + "They determine which MIDI channel is used to light up the keys.</html>");
61          panel.add(channels, "1, 1");
62          panel.add(lblInfo, "3, 1");
63  
64          container.add(new JLabel("Navigation Channel:"), x + ", " + y + ", right, default");
65          container.add(panel, x + 2 + ", " + y + ", " + (width - 2) + ", 1, fill, default");
66      }
67  
68      /**
69       * Loads the current settings from the user's preferences.
70       *
71       * @param preferences
72       *            the preferences to load from
73       */
74      void load(Preferences preferences) {
75          channels.setSelectedItem(preferences.getInt(NAV_CHANNEL_PREFERENCE, 3) + 1);
76      }
77  
78      /**
79       * Saves the current setting to the user's preferences.
80       * <p>
81       * Any registered preference listeners will be called.
82       *
83       * @param preferences
84       *            the preferences to save to
85       */
86      void save(Preferences preferences) {
87          preferences.put(NAV_CHANNEL_PREFERENCE, Integer.toString((Integer) channels.getSelectedItem() - 1));
88      }
89  }
