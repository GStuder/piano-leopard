3   import java.awt.Component;
4   import java.awt.event.ComponentAdapter;
5   import java.awt.event.ComponentEvent;
6   import java.util.concurrent.TimeUnit;
7   
8   import javax.swing.JScrollBar;
9   import javax.swing.SwingUtilities;
10  
11  import net.kreatious.pianoleopard.midi.sequencer.OutputModel;
12  import net.kreatious.pianoleopard.painter.layout.DefaultEventLayout;
13  import net.kreatious.pianoleopard.painter.layout.EventLayout;
14  
15  /**
16   * Provides control over the current song position
17   *
18   * @author Jay-R Studer
19   */
20  class CurrentPositionController {
21      private CurrentPositionController() {
22      }
23  
24      /**
25       * Constructs a view and associates it with its controller
26       */
27      static Component create(OutputModel outputModel) {
28          final JScrollBar scrollBar = new JScrollBar();
29          scrollBar.setBlockIncrement(convertTime(TimeUnit.SECONDS.toMicros(30)));
30          scrollBar.setUnitIncrement(convertTime(TimeUnit.SECONDS.toMicros(1)));
31  
32          // Update track sizes
33          outputModel.addStartListener(s -> scrollBar.setMaximum(convertTime(s.getSequence().getMicrosecondLength())));
34          scrollBar.addComponentListener(new ComponentAdapter() {
35              private final EventLayout layout = new DefaultEventLayout(scrollBar.getSize());
36  
37              @Override
38              public void componentResized(ComponentEvent e) {
39                  layout.setComponentDimensions(scrollBar.getSize());
40                  final long visibleTime = layout.getHighestVisibleTime(0) - layout.getLowestVisibleTime(0);
41                  scrollBar.setVisibleAmount(convertTime(visibleTime));
42              }
43          });
44  
45          // Update thumb positions
46          scrollBar.addAdjustmentListener(l -> {
47              if (l.getValueIsAdjusting()) {
48                  outputModel.setCurrentTime(convertValue(l.getValue()));
49              }
50          });
51          outputModel.addCurrentTimeListener(time -> SwingUtilities.invokeLater(() -> {
52              if (!scrollBar.getValueIsAdjusting()) {
53                  scrollBar.setValue(convertTime(time));
54              }
55          }));
56          return scrollBar;
57      }
58  
59      private static long convertValue(int time) {
60          return (long) time << 10;
61      }
62  
63      private static int convertTime(long time) {
64          // Rolls over after 50 days, provides millisecond accuracy
65          return (int) (time >> 10);
66      }
67  }
