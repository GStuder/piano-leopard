18  /**
19   * Provides the controller for altering which tracks are being practiced.
20   *
21   * @author Jay-R Studer
22   */
23  class PracticeTrackController {
24      private PracticeTrackController() {
25      }
26  
27      /**
28       * Constructs a view and associates it with its controller
29       */
30      static Component create(OutputModel outputModel) {
31          final JPanel panel = new JPanel(new FlowLayout(FlowLayout.LEFT));
32          outputModel.addStartListener(new StartListener(panel));
33          return panel;
34      }
35  
36      /**
37       * Reconstructs the panel's buttons when a new song is started.
38       *
39       * @author Jay-R Studer
40       */
41      private static final class StartListener implements Consumer<ParsedSequence> {
42          private final JPanel panel;
43          private static final Predicate<ParsedTrack> CONTAINS_NOTES = track -> track.getNotePairs(0, Long.MAX_VALUE)
44                  .iterator().hasNext();
45  
46          private StartListener(JPanel panel) {
47              this.panel = panel;
48          }
49  
50          @Override
51          public void accept(ParsedSequence sequence) {
52              panel.removeAll();
53              panel.add(new JLabel("Tracks:"));
54              sequence.getTracks().stream().filter(CONTAINS_NOTES)
55                      .forEach(track -> panel.add(createButton(sequence, track)));
56              panel.revalidate();
57          }
58  
59          private static Component createButton(ParsedSequence sequence, ParsedTrack track) {
60              final JToggleButton button = new JToggleButton();
61              button.setMargin(new Insets(8, 8, 8, 8));
62              button.addItemListener(e -> sequence.setTrackActive(track, e.getStateChange() == ItemEvent.SELECTED));
63              button.setSelected(sequence.getActiveTracks().contains(track));
64              button.addMouseListener(new ToggleListener(toggle -> sequence.setTrackActive(track, button.isSelected()
65                      ^ toggle)));
66              return button;
67          }
68      }
69  }
