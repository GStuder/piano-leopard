1   package net.kreatious.pianoleopard.painter.layout;
2   
3   import java.awt.Dimension;
4   import java.awt.Rectangle;
5   
6   import net.kreatious.pianoleopard.midi.event.NoteEvent;
7   
8   /**
9    * Provides the default horizontal layout strategy, laying out low notes on the
10   * left and high notes on the right.
11   *
12   * @author Jay-R Studer
13   */
14  class DefaultXNoteStrategy implements XNoteStrategy {
15      private static final int LOWEST_PLAYABLE_KEY = 36;
16      private static final int HIGHEST_PLAYABLE_KEY = 96;
17  
18      private float minXFactor;
19      private float normalWidth;
20  
21      @Override
22      public void layout(NoteEvent note, Rectangle rect) {
23          final NoteSize size = NoteSize.forPitch(note.getKey());
24          rect.x = (int) (normalWidth * (size.getXOffsetFactor() - minXFactor));
25          rect.width = (int) (normalWidth * size.getWidthFactor());
26      }
27  
28      @Override
29      public void setComponentDimensions(Dimension dimension) {
30          minXFactor = NoteSize.forPitch(LOWEST_PLAYABLE_KEY).getXOffsetFactor();
31  
32          final NoteSize highestSize = NoteSize.forPitch(HIGHEST_PLAYABLE_KEY);
33          normalWidth = dimension.width / (highestSize.getXOffsetFactor() - minXFactor + highestSize.getWidthFactor());
34      }
35  }
