1   package net.kreatious.pianoleopard.painter.layout;

3   import java.awt.Dimension;
4   import java.awt.Rectangle;
5   
6   import net.kreatious.pianoleopard.midi.event.PedalEvent;
7   
8   /**
9    * Provides the default layout strategy for pedal events. Lays out pedal events
10   * such that they cover the entire width of the component.
11   *
12   * @author Jay-R Studer
13   */
14  class DefaultXPedalStrategy implements XPedalStrategy {
15      private int width;
16  
17      @Override
18      public void layout(PedalEvent pedal, Rectangle rect) {
19          rect.x = 0;
20          rect.width = width;
21      }
22  
23      @Override
24      public void setComponentDimensions(Dimension dimension) {
25          width = dimension.width;
26      }
27  }
