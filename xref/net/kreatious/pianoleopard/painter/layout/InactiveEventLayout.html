3   import java.awt.Dimension;
4   import java.awt.Rectangle;
5   
6   import net.kreatious.pianoleopard.midi.event.EventPair;
7   import net.kreatious.pianoleopard.midi.event.NoteEvent;
8   
9   /**
10   * Provides the coordinates for drawing events not to be played by the user.
11   *
12   * @author Jay-R Studer
13   */
14  public class InactiveEventLayout extends DefaultEventLayout {
15      /**
16       * Constructs a new {@link InactiveEventLayout} with the initial specified
17       * component layout dimensions
18       *
19       * @param componentDimension
20       *            the inital component layout dimensions
21       */
22      public InactiveEventLayout(Dimension componentDimension) {
23          super(componentDimension);
24      }
25  
26      @Override
27      public void layoutNote(long currentTime, EventPair<NoteEvent> event, Rectangle rect) {
28          super.layoutNote(currentTime, event, rect);
29          rect.x += rect.width / 4;
30          rect.width -= rect.width / 2;
31      }
32  }
