1   package net.kreatious.pianoleopard.painter.layout;
2   
3   import java.awt.Dimension;
4   import java.awt.Rectangle;
5   
6   import net.kreatious.pianoleopard.midi.event.NoteEvent;
7   
8   /**
9    * Provides the strategy used to horizontally layout notes on screen
10   *
11   * @author Jay-R Studer
12   */
13  interface XNoteStrategy {
14  
15      /**
16       * Performs the horizontal layout operation.
17       * <p>
18       * Implementors are to modify the x and width values of {@code rect}.
19       *
20       * @param note
21       *            the note event to layout
22       * @param rect
23       *            output parameter storing the resulting on screen location of
24       *            the event.
25       */
26      void layout(NoteEvent note, Rectangle rect);
27  
28      /**
29       * Sets the component dimensions to layout within
30       *
31       * @param dimension
32       *            the new dimensions of the component
33       */
34      void setComponentDimensions(Dimension dimension);
35  
36      /**
37       * Decorates this strategy with the specified strategy.
38       *
39       * @param next
40       *            the next strategy to apply after this strategy is called
41       * @return a new decorated layout strategy
42       */
43      default XNoteStrategy thenApply(XNoteStrategy next) {
44          return new XNoteStrategy() {
45              private final XNoteStrategy xNoteStrategy = XNoteStrategy.this;
46  
47              @Override
48              public void setComponentDimensions(Dimension dimension) {
49                  xNoteStrategy.setComponentDimensions(dimension);
50                  next.setComponentDimensions(dimension);
51              }
52  
53              @Override
54              public void layout(NoteEvent note, Rectangle rect) {
55                  xNoteStrategy.layout(note, rect);
56                  next.layout(note, rect);
57              }
58          };
59      }
60  }
