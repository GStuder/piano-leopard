1   package net.kreatious.pianoleopard.painter.layout;
2   
3   import static java.util.stream.Collectors.toMap;
4   
5   import java.util.EnumMap;
6   import java.util.Map;
7   import java.util.stream.IntStream;
8   
9   /**
10   * Represents the relative corrections to apply in order to draw a note in the
11   * same location as it would appear on a synthesizer.
12   *
13   * @author JStuder
14   */
15  class NoteSize {
16      private enum Pitch {
17          C, C_SHARP, D, D_SHARP, E, F, F_SHARP, G, G_SHARP, A, A_SHARP, B;
18  
19          private static final Pitch[] VALUES = values();
20  
21          /**
22           * Translates a MIDI key (note) value into a pitch
23           *
24           * @param midiNote
25           *            the raw MIDI key (note) value to translate (0-127)
26           * @return the corresponding {@link Pitch}
27           */
28          public static Pitch fromMidiNote(int midiNote) {
29              return VALUES[midiNote % 12];
30          }
31      }
32  
33      private static final Map<Integer, NoteSize> NOTE_SIZES;
34      static {
35          final Map<Pitch, NoteSize> sizes = new EnumMap<>(Pitch.class);
36          sizes.put(Pitch.C, forWhiteKey(-6f));
37          sizes.put(Pitch.C_SHARP, forLeftBlackKey(-5f));
38          sizes.put(Pitch.D, forWhiteKey(-5f));
39          sizes.put(Pitch.D_SHARP, forRightBlackKey(-4f));
40          sizes.put(Pitch.E, forWhiteKey(-4f));
41          sizes.put(Pitch.F, forWhiteKey(-3f));
42          sizes.put(Pitch.F_SHARP, forLeftBlackKey(-2f));
43          sizes.put(Pitch.G, forWhiteKey(-2f));
44          sizes.put(Pitch.G_SHARP, forCenterBlackKey(-1f));
45          sizes.put(Pitch.A, forWhiteKey(-1f));
46          sizes.put(Pitch.A_SHARP, forRightBlackKey(0f));
47          sizes.put(Pitch.B, forWhiteKey(0f));
48  
49          NOTE_SIZES = IntStream.range(0, 128).boxed()
50                  .collect(toMap(i -> i, i -> new NoteSize(i, sizes.get(Pitch.fromMidiNote(i)))));
51      }
52  
53      private static final float BLACK_WIDTH_FACTOR = 1.0f / 3.0f;
54      private final float xOffsetFactor;
55      private final float widthFactor;
56  
57      private NoteSize(float xOffsetFactor, float widthFactor) {
58          this.xOffsetFactor = xOffsetFactor;
59          this.widthFactor = widthFactor;
60      }
61  
62      private NoteSize(int pitch, NoteSize keySize) {
63          xOffsetFactor = (float) (keySize.xOffsetFactor + Math.floor(pitch / 12.0) * 7.0);
64          widthFactor = keySize.widthFactor;
65      }
66  
67      private static NoteSize forWhiteKey(float xAdjustment) {
68          return new NoteSize(xAdjustment, 1.0f);
69      }
70  
71      private static NoteSize forLeftBlackKey(float xAdjustment) {
72          return new NoteSize(xAdjustment - BLACK_WIDTH_FACTOR * 2 / 3, BLACK_WIDTH_FACTOR);
73      }
74  
75      private static NoteSize forCenterBlackKey(float xAdjustment) {
76          return new NoteSize(xAdjustment - BLACK_WIDTH_FACTOR * 1 / 2, BLACK_WIDTH_FACTOR);
77      }
78  
79      private static NoteSize forRightBlackKey(float xAdjustment) {
80          return new NoteSize(xAdjustment - BLACK_WIDTH_FACTOR * 1 / 3, BLACK_WIDTH_FACTOR);
81      }
82  
83      static NoteSize forPitch(int pitch) {
84          return NOTE_SIZES.get(pitch);
85      }
86  
87      /**
88       * @return the normalized offset necessary to correctly position the note.
89       */
90      float getXOffsetFactor() {
91          return xOffsetFactor;
92      }
93  
94      /**
95       * @return the factor to apply to the width of the note
96       */
97      float getWidthFactor() {
98          return widthFactor;
99      }
100 }
