1   package net.kreatious.pianoleopard.painter.layout;
2   
3   import java.awt.Dimension;
4   import java.awt.Rectangle;
5   import java.util.concurrent.TimeUnit;
6   
7   import net.kreatious.pianoleopard.midi.event.Event;
8   import net.kreatious.pianoleopard.midi.event.EventPair;
9   
10  /**
11   * Lays out notes such that they move from the top of the screen downwards.
12   *
13   * @author Jay-R Studer
14   */
15  class DownwardsYStrategy implements YStrategy {
16      private static final int MICROS_PER_PIXEL = (int) TimeUnit.MILLISECONDS.toMicros(5);
17      private int playBarY;
18  
19      @Override
20      public void layout(long currentTime, EventPair<? extends Event> event, Rectangle rect) {
21          rect.y = (int) ((event.getOnTime() - currentTime) / MICROS_PER_PIXEL);
22          rect.height = (int) (event.getDuration() / MICROS_PER_PIXEL);
23          rect.y = playBarY - rect.y - rect.height;
24      }
25  
26      @Override
27      public void setComponentDimensions(Dimension dimension) {
28          playBarY = dimension.height / 2;
29      }
30  
31      @Override
32      public long getHighestVisibleTime(long currentTime) {
33          return currentTime + playBarY * MICROS_PER_PIXEL;
34      }
35  
36      @Override
37      public long getLowestVisibleTime(long currentTime) {
38          return currentTime - playBarY * MICROS_PER_PIXEL;
39      }
40  }
