3   import java.awt.Dimension;
4   import java.awt.Graphics;
5   import java.awt.Graphics2D;
6   import java.awt.event.ComponentAdapter;
7   import java.awt.event.ComponentEvent;
8   
9   import javax.swing.JPanel;
10  
11  import net.kreatious.pianoleopard.midi.ParsedSequence;
12  import net.kreatious.pianoleopard.midi.ParsedTrack;
13  import net.kreatious.pianoleopard.midi.sequencer.InputModel;
14  import net.kreatious.pianoleopard.midi.sequencer.OutputModel;
15  
16  /**
17   * Renders the currently playing sequence into a panel using double buffering.
18   *
19   * @author Jay-R Studer
20   */
21  public class PainterPanel {
22      private final class PainterPanelImpl extends JPanel {
23          private static final long serialVersionUID = 3647951920113354307L;
24          private final Painter painter = new Painter(new Dimension());
25  
26          PainterPanelImpl() {
27              setPreferredSize(new Dimension(1000, 500));
28              addComponentListener(new ComponentAdapter() {
29                  @Override
30                  public void componentResized(ComponentEvent e) {
31                      painter.setComponentDimensions(getSize());
32                  }
33              });
34          }
35  
36          @Override
37          public void paint(Graphics g) {
38              painter.paint((Graphics2D) g, currentTime, sequence, playedTrack);
39          }
40      }
41  
42      private final JPanel panel = new PainterPanelImpl();
43      private final ParsedTrack playedTrack;
44  
45      private volatile long currentTime;
46      private volatile ParsedSequence sequence = ParsedSequence.createEmpty();
47  
48      /**
49       * Constructor declared private to prevent direct instantiation by
50       * consumers.
51       */
52      private PainterPanel(ParsedTrack playedTrack) {
53          this.playedTrack = playedTrack;
54      }
55  
56      /**
57       * Constructs a new {@link PainterPanel} connected to the specified track
58       * containing events played by the user
59       *
60       * @param outputModel
61       *            the output model for events sent to the synthesizer
62       * @param inputModel
63       *            the input model for events played by the user
64       * @return a new instance of {@link PainterPanel}
65       */
66      public static JPanel create(OutputModel outputModel, InputModel inputModel) {
67          final PainterPanel result = new PainterPanel(inputModel);
68          outputModel.addCurrentTimeListener(result::setCurrentTime);
69          outputModel.addStartListener(result::setCurrentSequence);
70          return result.getPanel();
71      }
72  
73      /**
74       * Gets the panel associated with this component
75       *
76       * @return the panel owned by this component
77       */
78      private JPanel getPanel() {
79          return panel;
80      }
81  
82      /**
83       * @param currentTime
84       *            the current song time in microseconds
85       */
86      private void setCurrentTime(long currentTime) {
87          this.currentTime = currentTime;
88          panel.repaint();
89      }
90  
91      /**
92       * Sets the current sequence displayed by this panel
93       *
94       * @param sequence
95       *            the parsed sequence to set
96       */
97      private void setCurrentSequence(ParsedSequence sequence) {
98          this.sequence = sequence;
99      }
100 }
