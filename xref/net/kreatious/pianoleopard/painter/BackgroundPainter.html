1   package net.kreatious.pianoleopard.painter;
2   
3   import java.awt.BasicStroke;
4   import java.awt.Color;
5   import java.awt.Dimension;
6   import java.awt.Graphics2D;
7   import java.awt.Rectangle;
8   import java.awt.Stroke;
9   import java.util.stream.IntStream;
10  
11  import net.kreatious.pianoleopard.midi.event.EventPair;
12  import net.kreatious.pianoleopard.midi.event.NoteEvent;
13  import net.kreatious.pianoleopard.painter.layout.DefaultEventLayout;
14  import net.kreatious.pianoleopard.painter.layout.EventLayout;
15  
16  /**
17   * Paints the background into a graphics context
18   *
19   * @author Jay-R Studer
20   */
21  class BackgroundPainter {
22      private static final EventPair<NoteEvent>[] SHARP_NOTES = IntStream.range(0, 128)
23              .mapToObj(key -> new EventPair<>(new NoteEvent(key, true, 0), new NoteEvent(key, false, 0)))
24              .filter(pair -> pair.getOff().isSharp()).toArray(EventPair[]::new);
25  
26      private static final EventPair<NoteEvent>[] NATURAL_NOTES = IntStream.range(0, 128)
27              .mapToObj(key -> new EventPair<>(new NoteEvent(key, true, 0), new NoteEvent(key, false, 0)))
28              .filter(pair -> pair.getOff().isSharp() == false).toArray(EventPair[]::new);
29  
30      private static final Stroke STROKE = new BasicStroke();
31  
32      private final EventLayout layout;
33      private final Rectangle rect = new Rectangle();
34  
35      private int width;
36      private int height;
37  
38      /**
39       * Constructor declared private to prevent direct instantiation by
40       * consumers.
41       */
42      private BackgroundPainter(EventLayout layout, Dimension dimension) {
43          this.layout = layout;
44          width = dimension.width;
45          height = dimension.height;
46      }
47  
48      /**
49       * Constructs a new background painter.
50       *
51       * @param dimension
52       *            the initial component dimensions
53       * @return a new {@link BackgroundPainter}
54       */
55      static BackgroundPainter create(Dimension dimension) {
56          return new BackgroundPainter(new DefaultEventLayout(dimension), dimension);
57      }
58  
59      /**
60       * Paints the background into a graphics context
61       *
62       * @param graphics
63       *            the graphics context to paint into
64       */
65      void paint(Graphics2D graphics) {
66          paintNaturalNoteBackgrounds(graphics);
67          paintLinesBetweenNaturalNotes(graphics);
68          paintSharpNoteBackgrounds(graphics);
69      }
70  
71      private void paintNaturalNoteBackgrounds(Graphics2D graphics) {
72          graphics.setColor(new Color(160, 160, 160));
73          graphics.fillRect(0, 0, width, height);
74      }
75  
76      private void paintLinesBetweenNaturalNotes(Graphics2D graphics) {
77          graphics.setColor(Color.BLACK);
78          graphics.setStroke(STROKE);
79          for (final EventPair<NoteEvent> sharpNote : NATURAL_NOTES) {
80              layout.layoutNote(0, sharpNote, rect);
81              graphics.drawLine(rect.x, 0, rect.x, height);
82          }
83      }
84  
85      private void paintSharpNoteBackgrounds(Graphics2D graphics) {
86          graphics.setColor(new Color(96, 96, 96));
87          for (final EventPair<NoteEvent> sharpNote : SHARP_NOTES) {
88              layout.layoutNote(0, sharpNote, rect);
89              graphics.fillRect(rect.x, 0, rect.width, height);
90          }
91      }
92  
93      /**
94       * Resizes the layout to fit the specified component dimensions.
95       *
96       * @param dimension
97       *            the new component layout dimensions
98       */
99      void setComponentDimensions(Dimension dimension) {
100         layout.setComponentDimensions(dimension);
101         width = dimension.width;
102         height = dimension.height;
103     }
104 }
