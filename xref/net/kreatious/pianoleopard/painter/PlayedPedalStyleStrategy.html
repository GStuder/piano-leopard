3   import java.awt.BasicStroke;
4   import java.awt.Color;
5   import java.awt.Graphics2D;
6   import java.awt.Rectangle;
7   import java.awt.Stroke;
8   
9   import net.kreatious.pianoleopard.midi.event.EventPair;
10  import net.kreatious.pianoleopard.midi.event.PedalEvent;
11  
12  /**
13   * Provides the drawing method for pedal events being played by the user
14   *
15   * @author Jay-R Studer
16   */
17  class PlayedPedalStyleStrategy implements StyleStrategy<PedalEvent> {
18      private static final Stroke STROKE = new BasicStroke(2.0f);
19      private int smallestSeenChannel = 16;
20  
21      @Override
22      public void paint(Graphics2D graphics, EventPair<PedalEvent> event, Rectangle rect) {
23          // Certain keyboards send pedal events to multiple channels
24          if (event.getChannel() < smallestSeenChannel) {
25              smallestSeenChannel = event.getChannel();
26              return;
27          } else if (event.getChannel() != smallestSeenChannel) {
28              return;
29          }
30  
31          graphics.setColor(transparent(ActivePedalStyleStrategy.getPedalColor(event)));
32          graphics.setStroke(STROKE);
33          graphics.fillRect(rect.x, rect.y, rect.width, rect.height);
34  
35          graphics.setColor(ActivePedalStyleStrategy.getPedalColor(event));
36          graphics.drawRect(rect.x, rect.y, rect.width, rect.height);
37      }
38  
39      private static Color transparent(Color color) {
40          return new Color(color.getRGB() & 0x30FFFFFF, true);
41      }
42  }
