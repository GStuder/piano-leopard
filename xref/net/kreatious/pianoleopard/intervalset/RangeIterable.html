3   import java.util.Iterator;
4   import java.util.Optional;
5   import java.util.function.Predicate;
6   
7   class RangeIterable<V> implements Iterable<V> {
8       private final IntervalSet<V> set;
9       private final Interval range;
10      private final Optional<Entry<V>> first;
11  
12      RangeIterable(IntervalSet<V> intervalSet, long lowKey, long highKey) {
13          set = intervalSet;
14          range = new Interval(lowKey, highKey);
15          first = first();
16      }
17  
18      @Override
19      public Iterator<V> iterator() {
20          return new RangeIterator<>(set, range, first);
21      }
22  
23      private Optional<Entry<V>> first() {
24          final Predicate<Entry<V>> withinResult = current -> current.getKey().containsInterval(range);
25          final Predicate<Entry<V>> maximumIsAfterRangeStart = current -> current.getMaximum() >= range.getLow();
26  
27          Optional<Entry<V>> current = set.getRoot();
28          while (true) {
29              if (current.flatMap(Entry::getLeft).filter(maximumIsAfterRangeStart).isPresent()) {
30                  current = current.flatMap(Entry::getLeft);
31              } else if (current.filter(withinResult).isPresent()) {
32                  return current;
33              } else if (current.flatMap(Entry::getRight).isPresent()) {
34                  current = current.flatMap(Entry::getRight);
35              } else {
36                  return Optional.empty();
37              }
38          }
39      }
40  }
