49      /**
50       * Performs a left tree rotation about this entry
51       *
52       * @return the new root node to set, if any, otherwise an empty optional
53       */
54      private Optional<Entry<V>> rotateLeft() {
55          final Optional<Entry<V>> oldRight = right;
56          Optional<Entry<V>> newRoot = Optional.empty();
57  
58          right = oldRight.flatMap(Entry::getLeft);
59          oldRight.flatMap(Entry::getLeft).ifPresent(rightLeft -> rightLeft.parent = Optional.of(this));
60          oldRight.get().parent = parent;
61  
62          if (parent.flatMap(Entry::getLeft).equals(Optional.of(this))) {
63              parent.get().left = oldRight;
64          } else if (parent.isPresent()) {
65              parent.get().right = oldRight;
66          } else {
67              newRoot = oldRight;
68          }
69  
70          oldRight.get().left = Optional.of(this);
71          parent = oldRight;
72          augment();
73          return newRoot;
74      }
75  
76      /**
77       * Performs a right tree rotation about this entry
78       *
79       * @return the new root node to set, if any, otherwise an empty optional
80       */
81      private Optional<Entry<V>> rotateRight() {
82          final Optional<Entry<V>> oldLeft = left;
83          Optional<Entry<V>> newRoot = Optional.empty();
84  
85          left = oldLeft.flatMap(Entry::getRight);
86          oldLeft.flatMap(Entry::getRight).ifPresent(leftRight -> leftRight.parent = Optional.of(this));
87          oldLeft.get().parent = parent;
88  
89          if (parent.flatMap(Entry::getRight).equals(Optional.of(this))) {
90              parent.get().right = oldLeft;
91          } else if (parent.isPresent()) {
92              parent.get().left = oldLeft;
93          } else {
94              newRoot = oldLeft;
95          }
96  
97          oldLeft.get().right = Optional.of(this);
98          parent = oldLeft;
99          augment();
100         return newRoot;
101     }

103     /**
104      * Rebalances the tree by performing tree rotations
105      *
106      * @param root
107      *            the current root node of the tree
108      * @return the new root node of the tree
109      */
110     Optional<Entry<V>> rebalance(Optional<Entry<V>> root) {
111         red = true;
112 
113         final AtomicReference<Optional<Entry<V>>> currentRoot = new AtomicReference<>(root);
114         final Consumer<Entry<V>> assignToCurrentRoot = newRoot -> currentRoot.set(Optional.of(newRoot));
115         final Consumer<Entry<V>> setRed = current -> current.red = true;
116         final Consumer<Entry<V>> setBlack = current -> current.red = false;
117 
118         Optional<Entry<V>> node = Optional.of(this);
119         while (node.isPresent() && !node.equals(currentRoot.get())
120                 && node.flatMap(Entry::getParent).filter(Entry::isRed).isPresent()) {
121             final Optional<Entry<V>> nodeParent = node.flatMap(Entry::getParent);
122             final Optional<Entry<V>> parentParent = nodeParent.flatMap(Entry::getParent);
123             if (nodeParent.equals(parentParent.flatMap(Entry::getRight))) {
124                 final Optional<Entry<V>> parentParentLeft = parentParent.flatMap(Entry::getLeft);
125                 if (parentParentLeft.filter(Entry::isRed).isPresent()) {
126                     nodeParent.ifPresent(setBlack);
127                     parentParent.ifPresent(setRed);
128                     parentParentLeft.ifPresent(setBlack);
129                     node = parentParent;
130                 } else {
131                     if (node.equals(nodeParent.flatMap(Entry::getLeft))) {
132                         node = node.flatMap(Entry::getParent);
133                         node.flatMap(Entry::rotateRight).ifPresent(assignToCurrentRoot);
134                     }
135                     final Optional<Entry<V>> newParent = node.flatMap(Entry::getParent);
136                     final Optional<Entry<V>> newParentParent = newParent.flatMap(Entry::getParent);
137                     newParent.ifPresent(setBlack);
138                     newParentParent.ifPresent(setRed);
139                     newParentParent.flatMap(Entry::rotateLeft).ifPresent(assignToCurrentRoot);
140                 }
141             } else {
142                 final Optional<Entry<V>> parentParentRight = parentParent.flatMap(Entry::getRight);
143                 if (parentParentRight.filter(Entry::isRed).isPresent()) {
144                     nodeParent.ifPresent(setBlack);
145                     parentParent.ifPresent(setRed);
146                     parentParentRight.ifPresent(setBlack);
147                     node = parentParent;
148                 } else {
149                     if (node.equals(nodeParent.flatMap(Entry::getRight))) {
150                         node = node.flatMap(Entry::getParent);
151                         node.flatMap(Entry::rotateLeft).ifPresent(assignToCurrentRoot);
152                     }
153                     final Optional<Entry<V>> newParent = node.flatMap(Entry::getParent);
154                     final Optional<Entry<V>> newParentParent = newParent.flatMap(Entry::getParent);
155                     newParent.ifPresent(setBlack);
156                     newParentParent.ifPresent(setRed);
157                     newParentParent.flatMap(Entry::rotateRight).ifPresent(assignToCurrentRoot);
158                 }
159             }
160         }
161         augment();
162 
163         currentRoot.get().ifPresent(setBlack);
164         return currentRoot.get();
165     }

167     /**
168      * Removes this node from the tree.
169      *
170      * @param root
171      *            the current root node
172      * @return the new root node
173      */
174     Optional<Entry<V>> remove(Optional<Entry<V>> root) {
175         if (left.isPresent() && right.isPresent()) {
176             Entry<V> inOrderSuccessor = right.get();
177             while (inOrderSuccessor.left.isPresent()) {
178                 inOrderSuccessor = inOrderSuccessor.left.get();
179             }
180 
181             key = inOrderSuccessor.key;
182             values = inOrderSuccessor.values;
183             return inOrderSuccessor.remove(root);
184         }
185 
186         Optional<Entry<V>> currentRoot = root;
187         if (left.isPresent() || right.isPresent()) {
188             final Entry<V> onlyChild = left.isPresent() ? left.get() : right.get();
189 
190             onlyChild.parent = parent;
191             if (parent.isPresent()) {
192                 parent.get().insertNode(onlyChild);
193             } else {
194                 currentRoot = Optional.of(onlyChild);
195             }
196 
197             left = Optional.empty();
198             right = Optional.empty();
199             parent = Optional.empty();
200 
201             if (!red) {
202                 currentRoot = onlyChild.deletionRebalance(currentRoot);
203             }
204         } else if (parent.isPresent()) {
205             if (!red) {
206                 currentRoot = deletionRebalance(currentRoot);
207             }
208 
209             final Entry<V> p = parent.get();
210             if (p.left.equals(Optional.of(this))) {
211                 p.left = Optional.empty();
212             } else if (p.right.equals(Optional.of(this))) {
213                 p.right = Optional.empty();
214             }
215             parent = Optional.empty();
216         } else {
217             currentRoot = Optional.empty();
218         }
219 
220         return currentRoot;
221     }

223     /**
224      * Rebalances the tree by performing tree rotations, used after deleting.
225      *
226      * @param root
227      *            the current root node of the tree
228      * @return the new root node of the tree
229      */
230     private Optional<Entry<V>> deletionRebalance(Optional<Entry<V>> root) {
231         final AtomicReference<Optional<Entry<V>>> currentRoot = new AtomicReference<>(root);
232         final Consumer<Entry<V>> assignToCurrentRoot = newRoot -> currentRoot.set(Optional.of(newRoot));
233         final Consumer<Entry<V>> setRed = current -> current.red = true;
234         final Consumer<Entry<V>> setBlack = current -> current.red = false;
235 
236         Optional<Entry<V>> node = Optional.of(this);
237         while (!(node.equals(currentRoot.get()) || node.filter(Entry::isRed).isPresent())) {
238             final Optional<Entry<V>> nodeParent = node.flatMap(Entry::getParent);
239             final Optional<Entry<V>> nodeParentLeft = nodeParent.flatMap(Entry::getLeft);
240             final Optional<Entry<V>> nodeParentRight = nodeParent.flatMap(Entry::getRight);
241             if (node.equals(nodeParentRight)) {
242                 Optional<Entry<V>> sibling = nodeParentLeft;
243 
244                 if (sibling.filter(Entry::isRed).isPresent()) {
245                     sibling.ifPresent(setBlack);
246                     nodeParent.ifPresent(setRed);
247                     nodeParent.flatMap(Entry::rotateRight).ifPresent(assignToCurrentRoot);
248                     sibling = nodeParentLeft;
249                 }
250 
251                 final Optional<Entry<V>> siblingRight = sibling.flatMap(Entry::getRight);
252                 final Optional<Entry<V>> siblingLeft = sibling.flatMap(Entry::getLeft);
253                 if (siblingRight.filter(Entry::isRed).isPresent() || siblingLeft.filter(Entry::isRed).isPresent()) {
254                     if (!siblingLeft.filter(Entry::isRed).isPresent()) {
255                         siblingRight.ifPresent(setBlack);
256                         sibling.ifPresent(setRed);
257                         sibling.flatMap(Entry::rotateLeft).ifPresent(assignToCurrentRoot);
258                         sibling = nodeParentLeft;
259                     }
260                     sibling.ifPresent(s -> s.red = nodeParent.filter(Entry::isRed).isPresent());
261                     nodeParent.ifPresent(setRed);
262                     siblingLeft.ifPresent(setBlack);
263                     nodeParent.flatMap(Entry::rotateRight).ifPresent(assignToCurrentRoot);
264                     node = currentRoot.get();
265                 } else {
266                     sibling.ifPresent(setRed);
267                     node = nodeParent;
268                 }
269             } else {
270                 Optional<Entry<V>> sibling = nodeParentRight;
271                 if (sibling.filter(Entry::isRed).isPresent()) {
272                     sibling.ifPresent(setBlack);
273                     nodeParent.ifPresent(setRed);
274                     nodeParent.flatMap(Entry::rotateLeft).ifPresent(assignToCurrentRoot);
275                     sibling = nodeParentRight;
276                 }
277 
278                 final Optional<Entry<V>> siblingLeft = sibling.flatMap(Entry::getLeft);
279                 final Optional<Entry<V>> siblingRight = sibling.flatMap(Entry::getRight);
280                 if (siblingLeft.filter(Entry::isRed).isPresent() || siblingRight.filter(Entry::isRed).isPresent()) {
281                     if (!siblingRight.filter(Entry::isRed).isPresent()) {
282                         siblingLeft.ifPresent(setBlack);
283                         sibling.ifPresent(setRed);
284                         sibling.flatMap(Entry::rotateRight).ifPresent(assignToCurrentRoot);
285                         sibling = nodeParentRight;
286                     }
287                     sibling.ifPresent(s -> s.red = nodeParent.filter(Entry::isRed).isPresent());
288                     nodeParent.ifPresent(setBlack);
289                     siblingRight.ifPresent(setBlack);
290                     nodeParent.flatMap(Entry::rotateLeft).ifPresent(assignToCurrentRoot);
291                     node = currentRoot.get();
292                 } else {
293                     sibling.ifPresent(setRed);
294                     node = nodeParent;
295                 }
296             }
297         }
298         augment();
299 
300         node.ifPresent(setBlack);
301         return currentRoot.get();
302     }

304     private void augment() {
305         Optional<Entry<V>> node = Optional.of(this);
306         do {
307             node.get().maximum = Stream
308                     .concat(Stream.<Optional<Entry<V>>> of(node.flatMap(Entry::getLeft), node.flatMap(Entry::getRight))
309                             .filter(Optional::isPresent).map(child -> child.get().maximum),
310                             Stream.of(node.get().getKey().getHigh())).max(Long::compare).get();
311             node = node.flatMap(Entry::getParent);
312         } while (node.isPresent());
313     }

315     /**
316      * Finds the entry with the specified key.
317      *
318      * @return the entry with the specified key, otherwise empty.
319      */
320     Optional<Entry<V>> binarySearchExact(Interval searchKey) {
321         Optional<Entry<V>> current = Optional.of(this);
322         do {
323             final int comparison = searchKey.compareTo(current.get().getKey());
324             if (comparison < 0) {
325                 current = current.flatMap(Entry::getLeft);
326             } else if (comparison > 0) {
327                 current = current.flatMap(Entry::getRight);
328             } else {
329                 return current;
330             }
331         } while (current.isPresent());
332         return Optional.empty();
333     }

335     /**
336      * Finds the insertion position of a new entry with the specified key.
337      * <p>
338      * If the search key already exists, the corresponding entry is returned,
339      * otherwise the entry which should be the parent of the inserted value is
340      * returned.
341      *
342      * @return the insertion position for the specified interval, never empty.
343      */
344     Optional<Entry<V>> binarySearchInexact(Interval searchKey) {
345         Optional<Entry<V>> current = Optional.of(this);
346         while (true) {
347             final Optional<Entry<V>> previous = current;
348             final int comparison = searchKey.compareTo(current.get().getKey());
349             if (comparison < 0) {
350                 current = current.flatMap(Entry::getLeft);
351             } else if (comparison > 0) {
352                 current = current.flatMap(Entry::getRight);
353             } else {
354                 return current;
355             }
356 
357             if (!current.isPresent()) {
358                 return previous;
359             }
360         }
361     }

363     /**
364      * Inserts a entry as a child of this entry, without rebalancing the tree.
365      */
366     void insertNode(Entry<V> entry) {
367         if (entry.key.compareTo(getKey()) < 0) {
368             left = Optional.of(entry);
369         } else {
370             right = Optional.of(entry);
371         }
372     }

382     /**
383      * Adds a value to this entry.
384      * <p>
385      * Entries do not allow duplicates and act as a set.
386      *
387      * @param value
388      *            the value to add
389      * @return true if a value was added, false otherwise
390      */
391     boolean addValue(V value) {
392         Objects.requireNonNull(value);
393         if (values.contains(value)) {
394             return false;
395         }
396         values.add(value);
397         return true;
398     }

404         } else if (!(o instanceof Entry)) {
