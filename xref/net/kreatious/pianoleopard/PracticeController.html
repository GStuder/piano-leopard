3   import java.awt.Component;
4   import java.io.File;
5   import java.util.Optional;
6   
7   import javax.swing.JButton;
8   
9   import net.kreatious.pianoleopard.midi.ParsedSequence;
10  import net.kreatious.pianoleopard.midi.sequencer.OutputModel;
11  
12  /**
13   * Provides the controller for the practice action.
14   *
15   * @author Jay-R Studer
16   */
17  class PracticeController {
18      private PracticeController() {
19      }
20  
21      /**
22       * Constructs a view and associates it with its controller
23       */
24      static Component create(OutputModel outputModel) {
25          final JButton practiceButton = new JButton("Practice");
26          outputModel.addOpenListener(sequence -> {
27              updateText(practiceButton, sequence);
28              practiceButton.setEnabled(true);
29          });
30          practiceButton.addActionListener(e -> outputModel.start());
31          practiceButton.setEnabled(false);
32          return practiceButton;
33      }
34  
35      private static void updateText(JButton practiceButton, ParsedSequence sequence) {
36          final Optional<String> fileName = sequence.getFile().map(File::getName);
37          if (fileName.isPresent()) {
38              practiceButton.setText("Practice " + fileName.get());
39          } else {
40              practiceButton.setText("Practice");
41          }
42      }
43  }
