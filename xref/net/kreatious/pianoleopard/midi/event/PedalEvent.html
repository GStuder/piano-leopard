1   package net.kreatious.pianoleopard.midi.event;
2   
3   import javax.sound.midi.ShortMessage;
4   
5   /**
6    * Represents an event that is associated with a pedal.
7    *
8    * @author Jay-R Studer
9    */
10  public class PedalEvent extends Event {
11      private final Pedal pedal;
12      private final boolean on;
13      private final Slot slot;
14  
15      PedalEvent(ShortMessage message, long time) {
16          super(message, time);
17  
18          pedal = Pedal.lookup(message.getData1()).orElseThrow(
19                  () -> new IllegalArgumentException(message.getData1() + " is not a pedal message"));
20  
21          on = message.getData2() >= 64;
22          slot = new Slot(message.getChannel(), pedal);
23      }
24  
25      private PedalEvent(int channel, long time, Pedal pedal, boolean on, Slot slot) {
26          super(channel, time);
27          this.pedal = pedal;
28          this.on = on;
29          this.slot = slot;
30      }
31  
32      static boolean canCreate(ShortMessage message) {
33          return message.getCommand() == ShortMessage.CONTROL_CHANGE && message.getData1() >= 64
34                  && message.getData1() <= 67;
35      }
36  
37      /**
38       * @return the pedal associated with this event
39       */
40      public Pedal getPedal() {
41          return pedal;
42      }
43  
44      @Override
45      public boolean isOn() {
46          return on;
47      }
48  
49      @Override
50      public Slot getSlot() {
51          return slot;
52      }
53  
54      @Override
55      @SuppressWarnings("unchecked")
56      public PedalEvent createOff(long offTime) {
57          return new PedalEvent(getChannel(), offTime, pedal, false, slot);
58      }
59  }
