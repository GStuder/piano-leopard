1   package net.kreatious.pianoleopard.midi.event;
2   
3   import java.util.Optional;
4   
5   /**
6    * Represents the various foot pedal
7    *
8    * @author Jay-R Studer
9    */
10  public enum Pedal {
11      /**
12       * Sustain pedal.
13       * <p>
14       * While this pedal is pressed, played notes decay slowly. This sounds as if
15       * note off messages are ignored.
16       */
17      SUSTAIN(64),
18  
19      /**
20       * Sostenuto pedal.
21       * <p>
22       * When this pedal is pressed, only the notes currently being played are
23       * sustained. Commonly abbreviated as S.P.
24       */
25      SOSTENUTO(65),
26  
27      /**
28       * Portamento pedal.
29       * <p>
30       * While this pedal is pressed, notes glide continuously in pitch from one
31       * to the next.
32       */
33      PORTAMENTO(66),
34  
35      /**
36       * Soft pedal.
37       * <p>
38       * While this pedal is pressed, notes are much quieter.
39       */
40      SOFT(67);
41  
42      private final int data;
43  
44      private Pedal(int data) {
45          this.data = data;
46      }
47  
48      /**
49       * Returns the raw MIDI data value associated with a control change message.
50       * 
51       * @return the data1 value for a control change message using this pedal
52       */
53      public int getData() {
54          return data;
55      }
56  
57      /**
58       * Determines the pedal enum associated with a MIDI data value.
59       *
60       * @param data
61       *            the MIDI data1 value associated with a control change message
62       * @return an optional containing the pedal associated with the data, or
63       *         empty
64       */
65      public static Optional<Pedal> lookup(int data) {
66          switch (data) {
67          case 64:
68              return Optional.of(SUSTAIN);
69          case 65:
70              return Optional.of(PORTAMENTO);
71          case 66:
72              return Optional.of(SOSTENUTO);
73          case 67:
74              return Optional.of(SOFT);
75          default:
76              return Optional.empty();
77          }
78      }
79  }
