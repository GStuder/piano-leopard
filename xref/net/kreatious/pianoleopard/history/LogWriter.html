24  /**
25   * Provides logging of keys and songs played
26   *
27   * @author Jay-R Studer
28   */
29  class LogWriter {
30      private class LogReader {
31          private final HistoryVisitor visitor;
32          private long position;
33          private long timeOffset;
34  
35          private LogReader(HistoryVisitor visitor) {
36              this.visitor = visitor;
37          }
38  
39          /**
40           * Parses the log file.
41           * <p>
42           * Subsequent invocations of this method resume reading from the last
43           * position.
44           *
45           * @throws IOException
46           *             if an I/O error occurs
47           */
48          void parse() throws IOException {
49              final ByteBuffer readBuffer = ByteBuffer.allocateDirect(8192);
50              final long synchronizeAt = channel.size() - readBuffer.capacity();
51              while (position < synchronizeAt) {
52                  readBuffer.clear();
53                  channel.read(readBuffer, position);
54                  readBuffer.flip();
55                  parse(readBuffer);
56                  position += readBuffer.position();
57              }
58  
59              // Synchronize with writer to prevent concurrent logfile changes
60              synchronized (LogWriter.this) {
61                  do {
62                      readBuffer.clear();
63                      channel.read(readBuffer, position);
64                      readBuffer.flip();
65                      parse(readBuffer);
66                      position += readBuffer.position();
67                  } while (readBuffer.position() != 0);
68              }
69  
70              visitor.onParsingComplete();
71          }
72  
73          private void parse(ByteBuffer readBuffer) {
74              int nextPosition = readBuffer.position();
75              try {
76                  final byte[] hash = new byte[32];
77                  while (readBuffer.hasRemaining()) {
78                      nextPosition = readBuffer.position();
79                      switch (Message.getByValue(readBuffer.get())) {
80                      case KEY:
81                          visitor.onKey(readBuffer.get(), readBuffer.get(), readBuffer.get(),
82                                  readVariableLengthLong(readBuffer), timeOffset + readVariableLengthLong(readBuffer));
83                          break;
84                      case START:
85                          visitor.onStart(timeOffset + readVariableLengthLong(readBuffer));
86                          break;
87                      case FILE:
88                          readBuffer.get(hash);
89                          visitor.onFile(hash, timeOffset + readVariableLengthLong(readBuffer));
90                          break;
91                      case HEADER:
92                          // fall through
93                      case OFFSET_CHANGED:
94                          timeOffset = readVariableLengthLong(readBuffer);
95                          break;
96                      default:
97                          break;
98                      }
99                  }
100             } catch (final InvalidMidiDataException e) {
101                 e.printStackTrace();
102             } catch (final BufferUnderflowException e) {
103                 // Rewind when reading a partial message
104                 readBuffer.position(nextPosition);
105             }
106         }
107     }

109     @FunctionalInterface
110     private interface Payload {
111         void run() throws IOException;
112     }
113 
114     private final FileChannel channel;
115     private final ByteBuffer buffer = ByteBuffer.allocateDirect(128);
116     private final Map<HistoryVisitor, LogReader> listeners = new HashMap<>();
117     private long offset = System.currentTimeMillis();
118 
119     private LogWriter(FileChannel channel) {
120         this.channel = channel;
121     }
122 
123     /**
124      * Creates a new {@link LogWriter} that logs events generated by the user.
125      *
126      * @param channel
127      *            the channel to write to
128      * @param outputModel
129      *            the {@link OutputModel} to listen to
130      * @param inputModel
131      *            the {@link InputModel} to listen to
132      * @return a new {@link LogWriter} if the log file can be written to,
133      *         otherwise an empty object.
134      */
135     static Optional<LogWriter> create(FileChannel channel, OutputModel outputModel, InputModel inputModel) {
136         try {
137             final LogWriter logger = new LogWriter(channel);
138             logger.logHeader();
139             outputModel.addOpenListener(sequence -> {
140                 logger.flush();
141                 sequence.getFile().ifPresent(logger::logFile);
142             });
143             outputModel.addPlayListener(logger::logStart);
144             inputModel.addInputListener(logger::logKey);
145 
146             return Optional.of(logger);
147         } catch (final IOException e) {
148             e.printStackTrace();
149             return Optional.empty();
150         }
151     }
152 
153     /**
154      * Starts reading the entire log file using the specified visitor.
155      * <p>
156      * Before returning the specified visitor will visit the entire log file up
157      * to this point.
158      *
159      * @param visitor
160      *            the visitor associated with the listener
161      * @throws IOException
162      *             if an I/O error occurs
163      */
164     void startReading(HistoryVisitor visitor) throws IOException {
165         final LogReader reader = new LogReader(visitor);
166         reader.parse();
167 
168         synchronized (this) {
169             listeners.put(visitor, reader);
170         }
171     }
172 
173     /**
174      * Stops parsing notifications on the specified visitor.
175      *
176      * @param visitor
177      *            the visitor to stop
178      */
179     synchronized void stopReading(HistoryVisitor visitor) {
180         listeners.remove(visitor);
181     }
182 
183     private synchronized void writeMessage(Message message, Payload payload) {
184         try {
185             writeUnhandledMessage(message, payload);
186 
187             for (final LogReader manager : listeners.values()) {
188                 manager.parse();
189             }
190         } catch (final IOException e) {
191             e.printStackTrace();
192         }
193     }
194 
195     private synchronized void writeUnhandledMessage(Message message, Payload payload) throws IOException {
196         buffer.clear();
197         buffer.put(message.getHeaderByte());
198         payload.run();
199         buffer.flip();
200         channel.write(buffer);
201     }
202 
203     private synchronized void logHeader() throws IOException {
204         channel.position(channel.size());
205         writeUnhandledMessage(Message.HEADER, () -> writeVariableLengthLong(offset));
206     }
207 
208     private synchronized void logKey(Event event) {
209         if (event instanceof NoteEvent) {
210             final NoteEvent noteEvent = (NoteEvent) event;
211             writeMessage(Message.KEY, () -> {
212                 buffer.put((byte) ((event.isOn() ? ShortMessage.NOTE_ON : ShortMessage.NOTE_OFF) | noteEvent
213                         .getChannel()));
214                 buffer.put((byte) noteEvent.getKey());
215                 buffer.put((byte) noteEvent.getVelocity());
216                 writeVariableLengthLong(System.currentTimeMillis() - offset);
217                 writeVariableLengthLong(event.getTime());
218             });
219         } else if (event instanceof PedalEvent) {
220             final PedalEvent pedalEvent = (PedalEvent) event;
221             writeMessage(Message.KEY, () -> {
222                 buffer.put((byte) (ShortMessage.CONTROL_CHANGE | event.getChannel()));
223                 buffer.put((byte) pedalEvent.getPedal().getData());
224                 buffer.put((byte) (pedalEvent.isOn() ? 127 : 0));
225                 writeVariableLengthLong(System.currentTimeMillis() - offset);
226                 writeVariableLengthLong(event.getTime());
227             });
228         }
229     }
230 
231     private synchronized void logFile(File midi) {
232         writeMessage(Message.FILE, () -> {
233             buffer.put(Hashing.sha256().hashBytes(Files.toByteArray(midi)).asBytes());
234             writeVariableLengthLong(System.currentTimeMillis() - offset);
235         });
236     }
237 
238     private synchronized void logStart() {
239         offset = System.currentTimeMillis();
240         writeMessage(Message.OFFSET_CHANGED, () -> writeVariableLengthLong(offset));
241         writeMessage(Message.START, () -> writeVariableLengthLong(System.currentTimeMillis() - offset));
242     }
243 
244     private synchronized void writeVariableLengthLong(long value) {
245         long v = value;
246         while ((v & ~0x7F) != 0L) {
247             buffer.put((byte) (v & 0x7F));
248             v >>>= 7;
249         }
250         buffer.put((byte) (v | 0x80));
251     }
252 
253     private static long readVariableLengthLong(ByteBuffer buffer) {
254         long result = 0;
255         int position = 0;
256         byte v;
257         while (((v = buffer.get()) & 0x80) == 0x00) {
258             result += (long) v << position;
259             position += 7;
260         }
261         result += (v & 0x7FL) << position;
262         return result;
263     }
264 
265     private synchronized void flush() {
266         try {
267             channel.force(false);
268         } catch (final IOException e) {
269             e.printStackTrace();
270         }
271     }
